ctrl + alt + t      //Вызов терминала

cd folder_name      //Вход в директорий
mkdir folder_name   //Создание директория
ls -l               //вывод содержимого директория с использованием широкого формата
dir                 //---//---
tree                //выводит дерево файлов и директориев
clear               //Очищает содержимое консоли

touch file_name     //создает файл
echo >> file_name   //создает файл

rm file_name        //удаляет файл
rm folder_name -r   //рекурсивно удаляет директорий и всё внутри него
cp file_name path/new_file_name     //копирует файл с возможностью изменения имени. Пример: cp file1.html ~/file2.html
cp -R file_name path/new_file_name  //копирует директорий рекурсивно из текущей папки в новое место возможно с переименованием

mv folder_name path/new_folder_name           //перемещает директорий
echo -e "строка информации" >> имя_файла      //Заносит текстовую информацию в файл

cat имя_файла       //Отображает содержимое файла
nano имя_файла      //Открывает текстовый редактор для изменения файла.
                    //В редакторе:  ctrl + O - сохранить изменения;
                    //              ctrl + X - выйти из редактора
ls -l >> имя_файла  //направляет поток вывода в файл

gparted             //Программа редактор раделов
top                 //Программа системный монитор. Для выхода из неё q.
lsusb               //Выводит список подключенных USB устройств
lspci               //Выводит список PCI устройств
lspci | grep ключевое_слово   //Выводит список PCI устройств с фильтрацией по ключевому слову

Права доступа к файлу: (-rw-rw-r--)
-|rw-      | rw-    | r--
  владелец | группа | остальные
chmod -R 777 folder_name/     //Изменяет права на директорий
